Overview

Interface for documents in Tablatal format.

Tablatal is a 'list-type database format' by Devine Lu Linvega. See: wiki.xxiivv.com/#tablatal

require 'nodaire/tablatal'

doc = Nodaire::Tablatal.parse! <<~TBTL
  NAME    AGE   COLOR
  Erica   12    Opal
  Alex    23    Cyan
  Nike    34    Red
  Ruca    45    Grey
TBTL

doc.valid?    # true
doc.keys      # ["NAME", "AGE", "COLOR"]
doc.to_a.last # {"NAME"=>"Ruca", "AGE"=>"45", "COLOR"=>"Grey"}
doc.to_csv    # "NAME,AGE,COLOR\nErica,12,Opal\nAlex,23,..."

def to_csv
  CSV.generate do |csv|
    csv << keys
    data.each do |row|
      csv << keys.map { |key| row[key] }
    end
  end
end
